1    """ 
2    1. Loop through all (start from second element). 
3        a) Continually, if preceding element > current, shift (not swap) it into current one's place. 
4            (This is done towards the beginning of the array, until either array's end 
5            or a smaller element is encountered.) 
6        b) Replace last swapped element with the one which value was being used to shift elements. 
7     
8    NOTE: 
9        j >= 0 and array[j] > inserted, not array[j] > inserted and j >= 0 
10       as array[j] will throw when j is < 0. 
11    
12       Inserted element is placed at j + 1 instead of j to compensate 
13       for j's decrementation occurring in the while loop. 
14    
15   O(nÂ²) 
16   stable 
17   """
18   
19   
20   def sort(array):
21       for i in range(1, len(array)):
22           inserted = array[i]
23   
24           j = i - 1
25           while j >= 0 and array[j] > inserted:
26               array[j + 1] = array[j]
27               j -= 1
28   
29           array[j + 1] = inserted
30   
31       return array
32   
33   
34   print(sort([3, 2, 11, -1, 0]))
