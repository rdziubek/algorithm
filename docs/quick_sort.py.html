9    def sort(array, left, right):
10       i = left
11       j = right
12   
13       pivot = array[(left + right) // 2]
14   
15       while i <= j:
16           while array[i] < pivot:
17               i += 1
18           while array[j] > pivot:
19               j -= 1
20   
21           if i <= j:
22               array[i], array[j] = array[j], array[i]
23               i += 1
24               j -= 1
25   
26       if left < j:
27           sort(array, left, j)
28       if right > i:
29           sort(array, i, right)
30   
31       return array
