10   def cipher(key: list, text):

13       if len(text) % cols != 0:

18       matrix = [['' for _ in range(cols)] for _ in range(rows)]

23               if k < len(text):
24                   matrix[row][col] = text[k]
25                   k += 1
26               else:
27                   matrix[row][col] = ''

29       result = []
30       for col in range(cols):
31           for row in range(rows):
32               if matrix[row][key[col]] != '':
