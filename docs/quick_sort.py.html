1    """ 
2    1. Arrange elements relative to the pivot. 
3    2. Execute quick-sort on left and right sides of the array (pivot denotes the middle). 
4     
5    NOTE: 
6        `left < j` and `right > i` checks make sure there is something to further sort. 
7        `right` is inclusive. 
8     
9        Recursive calls use `j` and `i` indices instead of a middle index (used in merge-sort). 
10    
11   O(nÂ²) 
12   not stable 
13   """
14   
15   
16   def sort(array, left, right):
17       i = left
18       j = right
19   
20       pivot = array[(left + right) // 2]
21   
22       while i <= j:
23           while array[i] < pivot:
24               i += 1
25           while array[j] > pivot:
26               j -= 1
27   
28           if i <= j:
29               array[i], array[j] = array[j], array[i]
30               i += 1
31               j -= 1
32   
33       if left < j:
34           sort(array, left, j)
35       if right > i:
36           sort(array, i, right)
37   
38       return array
39   
40   
41   array = [3, 2, 11, -1, 0]
42   print(sort(array, 0, len(array) - 1))
