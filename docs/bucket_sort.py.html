bucket_sort.py 
1    """ 
2    1. Determine max value. 
3    2. Make buckets. 
4    3. Put into buckets. 
5    4. Sort each bucket. 
6    5. Print buckets. 
7     
8    NOTE: 
9        `array[i] / size` is not floored (by floor division, as opposed to 
10       the used integer cast from a real division) so as not to 
11       encounter rounding errors caused by truncating instead of 
12       rounding float numbers down. 
13    
14       `j` figures actual bucket's index out. 
15    
16       Elements 'overflowing' the bucket size are put into the last container. 
17    
18   complexity and stability depends on sort used 
19   """
20   
21   import insertion_sort
22   
23   
24   def sort(array, max_value):
25       span = max_value / len(array)
26   
27       buckets = []
28       for _ in range(len(array)):
29           buckets += [[]]
30   
31       for i in range(len(array)):
32           j = int(array[i] / span)
33           if j != len(array):
34               buckets[j] += [array[i]]
35           else:
36               buckets[j - 1] += [array[i]]
37   
38       for i in range(len(array)):
39           insertion_sort.sort(buckets[i])
40   
41       result = []
42       for i in range(len(array)):
43           result += buckets[i]
44       return result
45   
46   
47   array = [0.48, 0.27, 0.12, 0.21, 0.43, 0.25]
48   print(sort(array, max(array)))
