9    def sieve(limit):
10       array = [True] * (limit + 1)
11   
12       i = 2
13       while i * i <= limit:
14           if array[i] is True:
15               for j in range(i * i, limit + 1, i):
16                   array[j] = False
17           i += 1
18   
19       _result = []
20       for i in range(2, len(array)):
21           if array[i] is True:
22               _result.append(i)
23   
24       return _result
