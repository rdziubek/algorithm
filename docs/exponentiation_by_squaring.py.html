1    """ 
2    NOTE: 
3        Last bit is equal 1 when the modulus 2 of the number yields 1. 
4        When it is such, returned number is multiplied by its base 
5        besides squaring it. 
6     
7    O(log n) 
8    """
9    
10   
11   def exponentiate(base, power):
12       result = 1
13   
14       while power > 0:
15           if power % 2 == 1:
16               result *= base
17   
18           result *= result
19           power //= 2
20   
21       return result
22   
23   
24   print(exponentiate(2, 10))
