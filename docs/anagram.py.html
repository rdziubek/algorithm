7    def anagram(word1, word2):
8        if len(word1) != len(word2):
9            return False
10   
11       counts = [0] * 128
12       for i in range(len(word1)):
13           counts[ord(word1[i])] += 1
14       for i in range(len(word2)):
15           counts[ord(word2[i])] -= 1
16   
17       for i in range(128):
18           if counts[i] != 0:
19               return False
20
