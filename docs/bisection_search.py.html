7    def search(array, target):
8        left = 0
9        right = len(array) - 1
10   
11       while left <= right:
12           middle = (left + right) // 2
13   
14           if array[middle] < target:
15               left = middle + 1
16           elif array[middle] > target:
17               right = middle - 1
18           else:
19               return middle
20   
21       return -1
