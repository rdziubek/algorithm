1    """ 
2    1. Replace each character: 
3        (character + key) mod 26 
4     
5    NOTE: 
6        For the formula to be applied, ASCII needs to be zeroed out by subtracting the value 
7        of either 'a' or 'A' (1), depending on the case (2). (and later adding the value back) 
8     
9        (1) 
10           (character + key - 'a'|'A') mod 26 + 'a'|'A' 
11    
12       (2) 
13           'a' <= character <= 'z': 'a' 
14           'A' <= character <= 'Z': 'A' 
15    
16    
17       Following implementation considers single case only. 
18    
19   O(n) 
20   """
21   
22   
23   def cipher(key, text):
24       for i in range(len(text)):
25           candidate = ord(text[i]) + key
26   
27           if candidate > ord('z'):
28               candidate -= 26
29           elif candidate < ord('a'):
30               candidate += 26
31   
32           text[i] = chr(candidate)
33   
34       return text
